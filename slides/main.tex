%% ── 8. KIẾN TRÚC A: NGUYÊN LÝ ───────────────────────────────────────────────
\begin{frame}{Hướng A: Sinh mẫu bằng WGAN-GP}
  \begin{block}{Nguyên lý WGAN (Wasserstein GAN)}
    Thay thế hàm loss Jensen-Shannon bằng Wasserstein Distance để tránh
    biến mất gradient và mode collapse.
  \end{block}
  \vspace{10pt}
  \begin{itemize}
    \item \textbf{Gradient Penalty (GP)}: Ép Lipschitz constraint thông
          qua việc phạt định mức gradient trên trung bình các mẫu thực
          và giả.
    \item \textbf{Ổn định}: Giúp Generator học được các phân phối đặc
          trưng sắc nét hơn của traffic mạng.
  \end{itemize}
\end{frame}

          \item Kiến trúc: 4-layer Fully Connected.

    \item \textbf{Vấn đề của Hướng A}: Gần gũi với dữ liệu huấn luyện IDS

    \item \textbf{Nhu cầu An toàn}: Hệ thống cần được thử thách bởi các

  \end{itemize}
  \vspace{10pt}
  \begin{alertblock}{Giải pháp nâng cấp}
    Tích hợp phản hồi từ Detector vào hàm mất mát của GAN để tạo ra các
    biến thể đối kháng (Adversarial Examples).
  \end{alertblock}
\end{frame}

%% ── 12. ARMS RACE (KHÁI NIỆM) ────────────────────────────────────────────────
\begin{frame}{Hướng B: Tiến hóa qua Cuộc đua vũ trang (Arms Race)}
  \begin{itemize}
    \item \textbf{Round 1}: GAN đối đầu với Detector sơ khai.
    \item \textbf{Adversarial Training}: Detector học từ thất bại để trở
          nên mạnh mẽ hơn.
    \item \textbf{Round 2}: GAN tự nâng cấp để tìm ra sơ hở của phiên bản
          IDS đã được hardening.
  \end{itemize}
  \vspace{10pt}
  \begin{block}{Chiến thuật Bypass}
    GAN học cách điều chỉnh các đặc trưng phi hàm (Non-functional) để làm
    lệch hướng phán đoán của các hệ thống AI phòng thủ.
  \end{block}
\end{frame}

%% ── 13. AUXILIARY LOSS ───────────────────────────────────────────────────────
\begin{frame}{Hướng B: Hàm mất mát phối hợp (Auxiliary Loss)}
  Công thức tối ưu cho Generator nâng cấp:
  \[ \mathcal{L}_G^{Total} = \mathcal{L}_{WGAN}
     + \alpha \cdot \mathcal{L}_{adv}
     + \beta  \cdot \mathcal{L}_{constraint} \]
  \vspace{6pt}
  \begin{itemize}
    \item $\mathcal{L}_{adv}$: Phản hồi từ IDS, ép IDS phân loại nhầm là
          Benign.
    \item $\mathcal{L}_{constraint}$: Đảm bảo các đặc trưng Functional
          không bị biến đổi quá mức.
    \item \textbf{Trọng số $\alpha, \beta$}: Điều tiết giữa sức mạnh tấn
          công và tính thực tế của mẫu mạng.
      \begin{itemize}
        \item Thực nghiệm: $\alpha = 1.0$, $\beta = 0.5$ ---
              \textit{ưu tiên sức mạnh bypass trong giai đoạn Red Teaming}.
        \item Tăng $\beta$ khi muốn giữ tính mạng lưới cao hơn
              (trade-off bypass vs.\ realism).
      \end{itemize}
  \end{itemize}
\end{frame}

%% ── 14. PHÂN LỚP FEATURE ─────────────────────────────────────────────────────
\begin{frame}{Hướng B: Phân lớp Feature trong đối kháng}
  Chúng tôi chia 80 features thành:
  \begin{itemize}
    \item \textcolor{danger}{\textbf{Functional (16 đặc trưng):}}
      Protocol, SYN Flag, Header Length\ldots\ Nếu GAN thay đổi bừa bãi,
      traffic sẽ không chạy được trên hạ tầng mạng thật.
    \item \textcolor{detcolor}{\textbf{Non-functional (64 đặc trưng):}}
      IAT (Inter-arrival Time), Byte counts\ldots\ Đây là ``vùng sáng tạo''
      để GAN tìm kẽ hở bypass.
  \end{itemize}
  \begin{alertblock}{Chiến lược Constraint}
    Sử dụng MSE loss để khóa chặt 16 Functional features về gần dải giá trị
    của traffic DDoS thật $[P_1, P_{99}]$.
  \end{alertblock}
\end{frame}

%% ── 15. ADVERSARIAL TRAINING ─────────────────────────────────────────────────
\begin{frame}{Hướng B: Cơ chế gia cố IDS (Adversarial Training)}
  \begin{itemize}
    \item Lấy dữ liệu GAN Round 1 sinh ra (mẫu đã lừa được detector).
    \item Trộn với dữ liệu thực theo tỉ lệ 50--50.
    \item Huấn luyện lại bộ phân loại MLP ban đầu.
  \end{itemize}
  \vspace{10pt}
  \begin{exampleblock}{Kết quả Giai đoạn này}
    Mô hình IDS (Detector v2) sau khi gia cố sẽ có khả năng chống lại không
    chỉ DDoS thực mà cả các mẫu GAN sơ cấp.
  \end{exampleblock}
\end{frame}

%% ── 16. ĐỘ ĐO ĐÁNH GIÁ (Realistic) ──────────────────────────────────────────
\begin{frame}{Độ đo đánh giá Hướng A (Realistic)}
  \begin{itemize}
    \item \textbf{Wasserstein Distance (WD)}: Khoảng cách dịch chuyển tối
          thiểu giữa 2 phân phối.
    \item \textbf{MMD (Maximum Mean Discrepancy)}: So sánh độ lệch vector
          đặc trưng trong không gian kernel.
    \item \textbf{Validity Score}:
          \% mẫu giả có đặc trưng Functional nằm trong giới hạn thực thi
          mạng $[P_1, P_{99}]$.
  \end{itemize}
\end{frame}

%% ── 17. ĐỘ ĐO ĐÁNH GIÁ (Attack) ─────────────────────────────────────────────
\begin{frame}{Độ đo đánh giá Hướng B (Attack)}
  \begin{itemize}\setlength\itemsep{8pt}
    \item \textbf{FoolRate (Bypass Rate)}: Tỷ lệ \% mẫu DDoS sinh ra mà
          máy học phân loại thành Benign.
    \item \textbf{F1-Score}: Đánh giá khả năng nhận diện của Detector.
          Càng thấp nghĩa là GAN càng thành công.
    \item \textbf{Transferability}: Khả năng tấn công chéo vào các mô hình
          Black-box (Random Forest).
  \end{itemize}
\end{frame}
